#include <iostream>
using namespace std;

int main() {
    int N, first = 0, second = 1, next;

    cout << "Enter the number of terms: ";
    cin >> N;

    cout << "Fibonacci Series: ";

    for (int i = 1; i <= N; i++) {
        cout << first << " ";

        next = first + second;
        first = second;
        second = next;
    }

    return 0;
}
